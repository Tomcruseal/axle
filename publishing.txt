 edit project/Build.scala (change version)
 git tag

 git add project/Build.scala
 sbt publish-signed
 git commit -m "tagged v0.1-M12"
 git push --tags
 https://oss.sonatype.org/
 Staging Repositories
 find org.pingel repository
 close
 "closing axle 0.1-M12"
 wait
 find org.pingel repository
 release
 "releasing axle 0.1-M12"
 confirm at https://oss.sonatype.org/content/repositories/releases/org/pingel/
 edit project/Build.scala (change version)
 git add project/Build.scala
 git commit -m "back to 0.1-SNAPSHOT"
